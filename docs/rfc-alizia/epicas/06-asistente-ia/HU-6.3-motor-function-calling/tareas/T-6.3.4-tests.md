## Descripcion

Tests unitarios para el tool registry, el tool execution loop y los tool sets. Se mockea el AI client para simular respuestas del LLM con y sin tool calls, permitiendo testear el flujo completo sin depender de Azure OpenAI.

## Test cases

### Tool Registry (`tests/ai/tool_registry_test.go`)

**6.11: Registrar tool → disponible en el registry**
```go
func TestToolRegistry_Register(t *testing.T) {
    registry := ai.NewToolRegistry()

    registry.Register(providers.ToolDefinition{
        Name:        "test_tool",
        Description: "A test tool",
        Parameters:  json.RawMessage(`{"type":"object"}`),
    }, func(args string) (string, error) {
        return "ok", nil
    })

    defs := registry.GetDefinitions()
    assert.Equal(t, 1, len(defs))
    assert.Equal(t, "test_tool", defs[0].Name)
    assert.True(t, registry.HasTool("test_tool"))
    assert.Equal(t, 1, registry.Count())
}
```

**6.12: Ejecutar tool registrada → handler llamado con args correctos**
```go
func TestToolRegistry_Execute(t *testing.T) {
    registry := ai.NewToolRegistry()
    var receivedArgs string

    registry.Register(providers.ToolDefinition{
        Name: "greet",
    }, func(args string) (string, error) {
        receivedArgs = args
        return `{"message":"hello"}`, nil
    })

    result, err := registry.Execute("greet", `{"name":"Ana"}`)

    assert.NoError(t, err)
    assert.Equal(t, `{"message":"hello"}`, result)
    assert.Equal(t, `{"name":"Ana"}`, receivedArgs)
}
```

### Tool Execution Loop (`tests/ai/tool_loop_test.go`)

**6.14: Single tool call → handler ejecutado → respuesta final**
```go
func TestRunToolLoop_SingleToolCall(t *testing.T) {
    // Mock AI client: primera llamada retorna tool call, segunda retorna mensaje final
    client := &mockAIClient{
        responses: []ai.CompletionResponse{
            {
                ToolCalls: []providers.ToolCall{{
                    ID:       "call_1",
                    Function: struct{ Name, Arguments string }{"greet", `{"name":"Ana"}`},
                }},
                TokensUsed: 100,
            },
            {
                Content:    "Le dije hola a Ana.",
                TokensUsed: 50,
            },
        },
    }

    registry := ai.NewToolRegistry()
    registry.Register(providers.ToolDefinition{Name: "greet"}, func(args string) (string, error) {
        return `{"status":"ok"}`, nil
    })

    resp, err := ai.RunToolLoop(context.Background(), client, messages, registry)

    assert.NoError(t, err)
    assert.Equal(t, "Le dije hola a Ana.", resp.Message)
    assert.Equal(t, []string{"greet"}, resp.ToolsUsed)
    assert.Equal(t, 150, resp.TokensUsed)
}
```

```go
func TestRunToolLoop_ToolError(t *testing.T) {

    registry := ai.NewToolRegistry()
    registry.Register(providers.ToolDefinition{Name: "update"}, func(args string) (string, error) {
```

```go
}
```

**6.17: Parallel tool calls → todos ejecutados**
```go
func TestRunToolLoop_ParallelToolCalls(t *testing.T) {
    client := &mockAIClient{
        responses: []ai.CompletionResponse{
            {
                ToolCalls: []providers.ToolCall{
                    {ID: "call_1", Function: struct{ Name, Arguments string }{"tool_a", `{}`}},
                    {ID: "call_2", Function: struct{ Name, Arguments string }{"tool_b", `{}`}},
                    {ID: "call_3", Function: struct{ Name, Arguments string }{"tool_c", `{}`}},
                },
                TokensUsed: 200,
            },
            {
                Content:    "Ejecute las tres operaciones.",
                TokensUsed: 50,
            },
        },
    }

    registry := ai.NewToolRegistry()
    executed := []string{}
    for _, name := range []string{"tool_a", "tool_b", "tool_c"} {
        n := name
        registry.Register(providers.ToolDefinition{Name: n}, func(args string) (string, error) {
            executed = append(executed, n)
            return "ok", nil
        })
    }

    resp, err := ai.RunToolLoop(context.Background(), client, messages, registry)

    assert.NoError(t, err)
    assert.Equal(t, []string{"tool_a", "tool_b", "tool_c"}, resp.ToolsUsed)
    assert.Equal(t, []string{"tool_a", "tool_b", "tool_c"}, executed)
    assert.Equal(t, 250, resp.TokensUsed)
}
```

## Mock helpers

```go
// mockAIClient simula respuestas del LLM para tests
type mockAIClient struct {
    responses      []ai.CompletionResponse
    alwaysToolCall bool
    callCount      int
}

func (m *mockAIClient) SendCompletion(ctx context.Context, messages []providers.ChatMessage, tools []providers.ToolDefinition) (*ai.CompletionResponse, error) {
    m.callCount++
    if m.alwaysToolCall {
        return &ai.CompletionResponse{
            ToolCalls:  []providers.ToolCall{{ID: fmt.Sprintf("call_%d", m.callCount), Function: struct{ Name, Arguments string }{"loop_tool", `{}`}}},
            TokensUsed: 10,
        }, nil
    }
    if m.callCount-1 < len(m.responses) {
        return &m.responses[m.callCount-1], nil
    }
    return &ai.CompletionResponse{Content: "done"}, nil
}
```

## Notas

- Todos los tests son unitarios — no requieren Azure OpenAI ni base de datos
- El `mockAIClient` simula secuencias de respuestas predeterminadas para testear el flujo del loop
- Los tests cubren el happy path, errores de tools, limite de iteraciones y tool calls paralelos
- Se usa `testify/assert` para assertions (consistente con el resto del proyecto)
