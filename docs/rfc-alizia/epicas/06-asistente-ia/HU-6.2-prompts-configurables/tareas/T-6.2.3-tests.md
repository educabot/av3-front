// tests/ai/prompt_test.go
package ai_test

import (
    "encoding/json"
    "testing"

    "go-alizia/src/core/usecases/ai"
)

func TestRenderPrompt(t *testing.T) {
    tests := []struct {
        name     string
        template string
        values   map[string]string
        want     string
        wantErr  bool
        errMsg   string
    }{
        {
            // Test case 6.6: Template con placeholders → todos reemplazados
            name:     "all placeholders replaced",
            template: "Area: {area_name}. Periodo: {period}. Categorias: {categories}.",
            values: map[string]string{
                "area_name":  "Ciencias Exactas",
                "period":     "Marzo-Junio 2026",
                "categories": "Algebra, Geometria",
            },
            want:    "Area: Ciencias Exactas. Periodo: Marzo-Junio 2026. Categorias: Algebra, Geometria.",
            wantErr: false,
        },
        {
            // Test case 6.7: Template con placeholder sin valor → error
            name:     "missing placeholder value returns error",
            template: "Area: {area_name}. Docente: {teacher_name}. Disciplina: {subject}.",
            values: map[string]string{
                "area_name": "Humanidades",
            },
            wantErr: true,
            errMsg:  "missing placeholder values: teacher_name, subject",
        },
        {
            name:     "template without placeholders returns as-is",
            template: "Este es un texto sin placeholders.",
            values:   map[string]string{},
            want:     "Este es un texto sin placeholders.",
            wantErr:  false,
        },
        {
            name:     "multiline template with markdown",
            template: "# Titulo\n\nArea: {area_name}\n\n## Secciones\n- {section_1}\n- {section_2}",
            values: map[string]string{
                "area_name": "Ciencias Naturales",
                "section_1": "Biologia",
                "section_2": "Quimica",
            },
            want:    "# Titulo\n\nArea: Ciencias Naturales\n\n## Secciones\n- Biologia\n- Quimica",
            wantErr: false,
        },
    }

    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            got, err := ai.RenderPrompt(tt.template, tt.values)
            if tt.wantErr {
                if err == nil {
                    t.Fatalf("expected error but got nil")
                }
                if tt.errMsg != "" && err.Error() != tt.errMsg {
                    t.Fatalf("expected error %q, got %q", tt.errMsg, err.Error())
                }
                return
            }
            if err != nil {
                t.Fatalf("unexpected error: %v", err)
            }
            if got != tt.want {
                t.Fatalf("expected %q, got %q", tt.want, got)
            }
        })
    }
}

// Test case 6.8: System prompt con rol + contexto + instruccion
func TestSystemPromptBuilder_RoleContextInstruction(t *testing.T) {
    prompt := ai.NewSystemPromptBuilder().
        AddContext("Documento", "Titulo: Plan Anual\nArea: Ciencias").
        WithInstruction("Genera una estrategia metodologica.").
        Build()

    // Debe contener las tres secciones
    assertContains(t, prompt, "Sos Alizia")
    assertContains(t, prompt, "# Contexto")
    assertContains(t, prompt, "## Documento")
    assertContains(t, prompt, "Titulo: Plan Anual")
    assertContains(t, prompt, "# Instruccion")
    assertContains(t, prompt, "Genera una estrategia metodologica.")
}

// Test case 6.8 (complemento): System prompt con constraints
func TestSystemPromptBuilder_WithConstraints(t *testing.T) {
    prompt := ai.NewSystemPromptBuilder().
        WithInstruction("Genera contenido.").
        AddConstraint("Maximo 500 palabras").
        AddConstraint("Formato markdown").
        Build()

    assertContains(t, prompt, "# Restricciones")
    assertContains(t, prompt, "- Maximo 500 palabras")
    assertContains(t, prompt, "- Formato markdown")
}

func TestSystemPromptBuilder_CustomRole(t *testing.T) {
    customRole := "Sos un asistente formal para la provincia de Buenos Aires."
    prompt := ai.NewSystemPromptBuilder().
        WithRole(customRole).
        WithInstruction("Genera un informe.").
        Build()

    assertContains(t, prompt, customRole)
    assertNotContains(t, prompt, "Sos Alizia")
}

// Test case 6.10: PromptConfig cargado desde org config → template listo para render
func TestPromptConfig_ParseFromJSON(t *testing.T) {
    raw := `{
        "key": "problem_edge",
        "ai_prompt": "Genera un eje para {area_name}.",
        "output_schema": {"type": "string"}
    }`

    var cfg ai.PromptConfig
    if err := json.Unmarshal([]byte(raw), &cfg); err != nil {
        t.Fatalf("failed to parse PromptConfig: %v", err)
    }

    if cfg.Key != "problem_edge" {
        t.Fatalf("expected key 'problem_edge', got %q", cfg.Key)
    }
    if cfg.AIPrompt != "Genera un eje para {area_name}." {
        t.Fatalf("unexpected ai_prompt: %q", cfg.AIPrompt)
    }
    if cfg.OutputSchema == nil {
        t.Fatal("expected output_schema to be present")
    }

    // Verificar que el template se puede renderizar
    result, err := ai.RenderPrompt(cfg.AIPrompt, map[string]string{
        "area_name": "Ciencias Exactas",
    })
    if err != nil {
        t.Fatalf("unexpected error: %v", err)
    }
    if result != "Genera un eje para Ciencias Exactas." {
        t.Fatalf("unexpected result: %q", result)
    }
}

// --- helpers ---

func assertContains(t *testing.T, s, substr string) {
    t.Helper()
    if !contains(s, substr) {
        t.Errorf("expected string to contain %q, got:\n%s", substr, s)
    }
}

func assertNotContains(t *testing.T, s, substr string) {
    t.Helper()
    if contains(s, substr) {
        t.Errorf("expected string NOT to contain %q, got:\n%s", substr, s)
    }
}

func contains(s, substr string) bool {
    return len(s) >= len(substr) && searchString(s, substr)
}

func searchString(s, substr string) bool {
    for i := 0; i <= len(s)-len(substr); i++ {
        if s[i:i+len(substr)] == substr {
            return true
        }
    }
    return false
}

## Notas

- Tests puros, sin dependencias externas — no requieren Azure OpenAI ni base de datos
- Se usa el patron table-driven para `TestRenderPrompt` (multiples escenarios en una sola funcion)
- Los helpers `assertContains`/`assertNotContains` simplifican las verificaciones de strings
- El test de `PromptConfig` valida tanto el parsing de JSON como la integracion con `RenderPrompt`
- La validacion de JSON Schema sobre salida de IA (test cases 6.9/6.10 originales) se testea a nivel de integracion con el AI Provider (HU-6.1) — aca se verifica la estructura de config
