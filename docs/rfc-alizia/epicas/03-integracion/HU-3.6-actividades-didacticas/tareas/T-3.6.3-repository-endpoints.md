## Descripcion

Implementar el repository GORM que satisface `ActivityProvider` y los endpoints HTTP para crear y listar actividades didacticas. El listado soporta filtro opcional por momento de clase via query param.

## Implementacion

### `src/repositories/activity_repository.go`

```go
package repositories

import (
    "context"

    "github.com/educabot/demo-alizia/src/core/entities"
    "github.com/educabot/demo-alizia/src/core/providers"
    "gorm.io/gorm"
)

type activityRepository struct {
    db *gorm.DB
}

func NewActivityRepository(db *gorm.DB) providers.ActivityProvider {
    return &activityRepository{db: db}
}

func (r *activityRepository) Create(ctx context.Context, activity *entities.Activity) (*entities.Activity, error) {
    err := r.db.WithContext(ctx).Create(activity).Error
    return activity, err
}

func (r *activityRepository) List(ctx context.Context, orgID uint, moment *entities.ClassMoment) ([]entities.Activity, error) {
    var activities []entities.Activity
    query := r.db.WithContext(ctx).Where("organization_id = ?", orgID)
    if moment != nil {
        query = query.Where("moment = ?", *moment)
    }
    err := query.Order("moment, name").Find(&activities).Error
    return activities, err
}

func (r *activityRepository) ListByMoment(ctx context.Context, orgID uint, moment entities.ClassMoment) ([]entities.Activity, error) {
    return r.List(ctx, orgID, &moment)
}
```

### Handler

```go
type ActivityHandler struct {
    provider providers.ActivityProvider
}

func (h *ActivityHandler) Create(c *gin.Context) {
    var req struct {
        Moment          string  `json:"moment" binding:"required,oneof=apertura desarrollo cierre"`
        Name            string  `json:"name" binding:"required"`
        Description     *string `json:"description"`
        DurationMinutes *int    `json:"duration_minutes"`
    }
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(400, gin.H{"error": err.Error()})
        return
    }

    activity := &entities.Activity{
        OrganizationID:  currentOrgID(c),
        Moment:          entities.ClassMoment(req.Moment),
        Name:            req.Name,
        Description:     req.Description,
        DurationMinutes: req.DurationMinutes,
    }

    result, err := h.provider.Create(c, activity)
    if err != nil {
        c.JSON(500, gin.H{"error": err.Error()})
        return
    }
    c.JSON(201, result)
}

func (h *ActivityHandler) List(c *gin.Context) {
    orgID := currentOrgID(c)

    var momentFilter *entities.ClassMoment
    if m := c.Query("moment"); m != "" {
        moment := entities.ClassMoment(m)
        momentFilter = &moment
    }

    activities, err := h.provider.List(c, orgID, momentFilter)
    if err != nil {
        c.JSON(500, gin.H{"error": err.Error()})
        return
    }
    c.JSON(200, activities)
}
```

### Registro en `mapping.go`

```go
// Activities
activities := api.Group("/activities")
activities.POST("", activityHandler.Create)
activities.GET("", activityHandler.List)
```

## Notas

- El filtro `?moment=apertura` es opcional. Sin el, retorna todas las actividades de la organizacion.
- La validacion `oneof=apertura desarrollo cierre` en el binding de gin asegura que solo se acepten momentos validos.
- El `Order("moment, name")` agrupa las actividades por momento y luego las ordena alfabeticamente dentro de cada grupo.
- Multi-tenancy: `currentOrgID(c)` extrae el organization_id del contexto de autenticacion, asegurando que cada org solo vea sus actividades.
