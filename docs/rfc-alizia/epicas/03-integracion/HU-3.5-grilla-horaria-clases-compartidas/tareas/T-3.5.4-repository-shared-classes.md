# T-3.5.4: Repository y queries de shared classes

**Historia:** HU-3.5 — Grilla horaria y clases compartidas
**Tipo:** Backend
**Estado:** ⬜ Pendiente

---

## Descripcion

Implementar el repository GORM que satisface `TimeSlotProvider`. Incluye las queries clave para listar slots de un curso, detectar clases compartidas y calcular que numeros de clase son compartidos para un course_subject dado.

## Implementacion

### `src/repositories/time_slot_repository.go`

```go
package repositories

import (
    "context"

    "github.com/educabot/demo-alizia/src/core/entities"
    "github.com/educabot/demo-alizia/src/core/providers"
    "gorm.io/gorm"
)

type timeSlotRepository struct {
    db *gorm.DB
}

func NewTimeSlotRepository(db *gorm.DB) providers.TimeSlotProvider {
    return &timeSlotRepository{db: db}
}

func (r *timeSlotRepository) Create(ctx context.Context, slot *entities.TimeSlot) (*entities.TimeSlot, error) {
    err := r.db.WithContext(ctx).Create(slot).Error
    if err != nil {
        return nil, err
    }
    // Recargar con preload
    return r.getByID(ctx, slot.ID)
}

func (r *timeSlotRepository) ListByCourse(ctx context.Context, courseID uint) ([]entities.TimeSlot, error) {
    var slots []entities.TimeSlot
    err := r.db.WithContext(ctx).
        Where("course_id = ?", courseID).
        Preload("Subjects").
        Preload("Subjects.CourseSubject").
        Order("day_of_week, start_time").
        Find(&slots).Error
    return slots, err
}

func (r *timeSlotRepository) GetSharedClasses(ctx context.Context, courseID uint) ([]providers.SharedClassInfo, error) {
    // Query: slots con 2+ subjects
    var results []providers.SharedClassInfo
    err := r.db.WithContext(ctx).Raw(`
        SELECT ts.day_of_week, ts.start_time, ts.end_time,
               array_agg(cs.id) AS course_subject_ids
        FROM time_slots ts
        JOIN time_slot_subjects tss ON tss.time_slot_id = ts.id
        JOIN course_subjects cs ON cs.id = tss.course_subject_id
        WHERE ts.course_id = ?
        GROUP BY ts.id, ts.day_of_week, ts.start_time, ts.end_time
        HAVING count(*) > 1
    `, courseID).Scan(&results).Error
    return results, err
}

func (r *timeSlotRepository) GetSharedClassNumbers(ctx context.Context, courseSubjectID uint, totalClasses int) ([]int, error) {
    // Paso 1: obtener posiciones semanales del subject y cuales son compartidas
    type weeklySlot struct {
        WeeklyPosition int
        IsShared       bool
    }
    var slots []weeklySlot
    err := r.db.WithContext(ctx).Raw(`
        WITH subject_slots AS (
            SELECT ts.id AS slot_id,
                   ROW_NUMBER() OVER (ORDER BY ts.day_of_week, ts.start_time) AS weekly_position
            FROM time_slots ts
            JOIN time_slot_subjects tss ON tss.time_slot_id = ts.id
            WHERE tss.course_subject_id = ?
        ),
        slot_counts AS (
            SELECT ss.weekly_position, ss.slot_id,
                   (SELECT count(*) FROM time_slot_subjects WHERE time_slot_id = ss.slot_id) AS subject_count
            FROM subject_slots ss
        )
        SELECT weekly_position, (subject_count > 1) AS is_shared
        FROM slot_counts
        ORDER BY weekly_position
    `, courseSubjectID).Scan(&slots).Error
    if err != nil {
        return nil, err
    }

    // Paso 2: proyectar el patron semanal sobre totalClasses
    if len(slots) == 0 {
        return []int{}, nil
    }
    classesPerWeek := len(slots)
    var sharedNumbers []int
    for classNum := 1; classNum <= totalClasses; classNum++ {
        weekPos := (classNum - 1) % classesPerWeek
        if slots[weekPos].IsShared {
            sharedNumbers = append(sharedNumbers, classNum)
        }
    }
    return sharedNumbers, nil
}

func (r *timeSlotRepository) getByID(ctx context.Context, id uint) (*entities.TimeSlot, error) {
    var slot entities.TimeSlot
    err := r.db.WithContext(ctx).
        Preload("Subjects").
        Preload("Subjects.CourseSubject").
        First(&slot, id).Error
    return &slot, err
}
```

## Notas

- La query de `GetSharedClasses` usa `GROUP BY` + `HAVING count(*) > 1` para encontrar slots con multiples subjects, tal como se define en el diseno tecnico de HU-3.5.
- `GetSharedClassNumbers` usa un enfoque de dos pasos: primero obtiene el patron semanal con SQL, luego lo proyecta en Go sobre el total de clases. Esto es mas eficiente que generar la serie completa en SQL.
- `array_agg` de PostgreSQL devuelve un array que GORM puede escanear en un slice de uint (puede requerir `pgtype` o scan custom dependiendo de la version de GORM).
- El Preload encadenado `Subjects.CourseSubject` permite resolver la informacion completa de la disciplina en una sola llamada.
