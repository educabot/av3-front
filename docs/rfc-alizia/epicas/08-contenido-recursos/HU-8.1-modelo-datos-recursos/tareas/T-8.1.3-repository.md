# T-8.1.3: Repository GORM

**Historia:** HU-8.1 — Modelo de datos de recursos
**Tipo:** Backend
**Estado:** ⬜ Pendiente

---

## Descripción

Implementar los repositories GORM que cumplen las interfaces de provider para resource_types, organization_resource_types y resources.

## Implementación

```go
// src/repositories/resource_repository.go
package repositories

type ResourceRepository struct {
    db *gorm.DB
}

func (r *ResourceRepository) GetByID(id int) (*entities.Resource, error) {
    var resource entities.Resource
    err := r.db.
        Preload("ResourceType").
        Preload("CourseSubject").
        Preload("CourseSubject.Subject").
        Preload("Creator").
        First(&resource, id).Error
    return &resource, err
}

func (r *ResourceRepository) GetByOrgID(orgID int, filters providers.ResourceFilters) ([]entities.Resource, error) {
    query := r.db.
        Preload("ResourceType").
        Preload("Creator").
        Where("organization_id = ?", orgID)

    if filters.ResourceTypeID != nil {
        query = query.Where("resource_type_id = ?", *filters.ResourceTypeID)
    }
    if filters.CourseSubjectID != nil {
        query = query.Where("course_subject_id = ?", *filters.CourseSubjectID)
    }
    if filters.SubjectID != nil {
        query = query.Joins("JOIN course_subjects cs ON cs.id = resources.course_subject_id").
            Where("cs.subject_id = ?", *filters.SubjectID)
    }
    if filters.Status != nil {
        query = query.Where("status = ?", *filters.Status)
    }
    if filters.CreatedBy != nil {
        query = query.Where("created_by = ?", *filters.CreatedBy)
    }

    query = query.Order("created_at DESC").Limit(filters.Limit).Offset(filters.Offset)

    var resources []entities.Resource
    err := query.Find(&resources).Error
    return resources, err
}

// ResourceTypeRepository
type ResourceTypeRepository struct {
    db *gorm.DB
}

func (r *ResourceTypeRepository) GetAvailableForOrg(orgID int) ([]entities.ResourceType, error) {
    var types []entities.ResourceType
    err := r.db.
        Where("organization_id IS NULL OR organization_id = ?", orgID).
        Order("name ASC").
        Find(&types).Error
    return types, err
}
```

## Notas

- El filtro por SubjectID hace un JOIN con course_subjects — no es FK directa
- Los preloads de Resource incluyen ResourceType y Creator para la vista de biblioteca
- El orden por defecto es `created_at DESC` (más recientes primero)
- `GetAvailableForOrg` combina tipos públicos (org NULL) y privados de la org
