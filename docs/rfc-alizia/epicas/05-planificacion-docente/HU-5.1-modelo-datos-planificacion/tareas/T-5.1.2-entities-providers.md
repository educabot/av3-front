# T-5.1.2: Entities y providers

**Historia:** HU-5.1 — Modelo de datos planificacion
**Tipo:** Backend
**Estado:** ⬜ Pendiente

---

## Descripcion

Crear las entities Go y las interfaces de provider para teacher_lesson_plans, lesson_plan_activities, lesson_plan_topics y lesson_plan_moment_fonts.

## Implementacion

### Entities

```go
// src/core/entities/teacher_lesson_plan.go

type LessonPlanStatus string

const (
    LessonPlanStatusPending    LessonPlanStatus = "pending"
    LessonPlanStatusInProgress LessonPlanStatus = "in_progress"
    LessonPlanStatusPublished  LessonPlanStatus = "published"
)

type SourceType string

const (
    SourceTypeResource SourceType = "resource"
    SourceTypeCustom   SourceType = "custom"
)

type ResourcesMode string

const (
    ResourcesModeGlobal    ResourcesMode = "global"
    ResourcesModePerMoment ResourcesMode = "per_moment"
)

type TeacherLessonPlan struct {
    ID                       uint             `gorm:"primaryKey" json:"id"`
    OrganizationID           uint             `json:"organization_id"`
    CourseSubjectID          uint             `json:"course_subject_id"`
    CoordinationDocumentID   uint             `json:"coordination_document_id"`
    CoordDocClassID          uint             `json:"coord_doc_class_id"`
    ClassNumber              int              `json:"class_number"`
    Title                    string           `json:"title"`
    Objective                *string          `json:"objective"`
    KnowledgeContent       *string          `json:"knowledge_content"`
    DidacticStrategies     *string          `json:"didactic_strategies"`
    ClassFormat            *string          `json:"class_format"`
    CustomInstruction      *string          `json:"custom_instruction"`
    ResourcesMode          *ResourcesMode   `json:"resources_mode"`
    Status                   LessonPlanStatus `json:"status"`
    Proposal                 *string          `json:"proposal"`
    SourceType               *SourceType      `json:"source_type"`
    SourceReference          *string          `json:"source_reference"`
    CreatedAt                time.Time        `json:"created_at"`
    UpdatedAt                time.Time        `json:"updated_at"`

    // Relations
    CourseSubject          *CourseSubject          `gorm:"foreignKey:CourseSubjectID" json:"course_subject,omitempty"`
    CoordinationDocument   *CoordinationDocument   `gorm:"foreignKey:CoordinationDocumentID" json:"coordination_document,omitempty"`
    CoordDocClass          *CoordDocClass           `gorm:"foreignKey:CoordDocClassID" json:"coord_doc_class,omitempty"`
    Activities             []LessonPlanActivity     `gorm:"foreignKey:TeacherLessonPlanID" json:"activities,omitempty"`
    Topics             []LessonPlanTopic        `gorm:"foreignKey:LessonPlanID" json:"topics,omitempty"`
    MomentFonts        []LessonPlanMomentFont   `gorm:"foreignKey:LessonPlanID" json:"moment_fonts,omitempty"`
}

type LessonPlanActivity struct {
    ID                   uint        `gorm:"primaryKey" json:"id"`
    TeacherLessonPlanID  uint        `json:"teacher_lesson_plan_id"`
    Moment               ClassMoment `json:"moment"`
    ActivityID           uint        `json:"activity_id"`
    CustomNote           *string     `json:"custom_note"`
    Order                int         `json:"order"`

    // Relations
    Activity *Activity `gorm:"foreignKey:ActivityID" json:"activity,omitempty"`
}

type LessonPlanTopic struct {
    ID           uint `gorm:"primaryKey" json:"id"`
    LessonPlanID uint `json:"lesson_plan_id"`
    TopicID      uint `json:"topic_id"`

    Topic *Topic `gorm:"foreignKey:TopicID" json:"topic,omitempty"`
}

type LessonPlanMomentFont struct {
    ID           uint         `gorm:"primaryKey" json:"id"`
    LessonPlanID uint         `json:"lesson_plan_id"`
    Moment       *ClassMoment `json:"moment"`  // nil = global
    FontID       uint         `json:"font_id"`

    Font *Font `gorm:"foreignKey:FontID" json:"font,omitempty"`
}
```

### Provider

```go
// src/core/providers/teacher_lesson_plan_provider.go

type TeacherLessonPlanProvider interface {
    Create(plan *entities.TeacherLessonPlan) error
    GetByID(id uint) (*entities.TeacherLessonPlan, error)
    GetByCourseSubject(courseSubjectID uint) ([]entities.TeacherLessonPlan, error)
    Update(plan *entities.TeacherLessonPlan) error
    Delete(id uint) error
    AddActivity(activity *entities.LessonPlanActivity) error
    RemoveActivity(planID uint, activityID uint, moment entities.ClassMoment) error
    GetActivitiesByPlan(planID uint) ([]entities.LessonPlanActivity, error)
    AddTopic(topic *entities.LessonPlanTopic) error
    RemoveTopic(planID uint, topicID uint) error
    GetTopicsByPlan(planID uint) ([]entities.LessonPlanTopic, error)
    AddMomentFont(mf *entities.LessonPlanMomentFont) error
    RemoveMomentFont(id uint) error
    GetMomentFontsByPlan(planID uint) ([]entities.LessonPlanMomentFont, error)
}
```

## Notas

- `ClassMoment` ya existe en entities desde HU-3.6
- Los punteros (*string, *SourceType) indican campos nullable
- `Font` entity viene de Épica 8 (HU-8.1) — `LessonPlanMomentFont` referencia fonts de la org
- `LessonPlanTopic` permite trazar qué topics del doc de coordinación se cubren en cada lesson plan
- Puntero en `Moment` (*ClassMoment) indica nullable — NULL = font global para todo el plan
